print: 'PRINT' STRING | INT | DECIMAL
eg.
PRINT "Hello"

declaration: TYPE NAME '=' STRING | INT | DECIMAL | expression
eg.
int a = 12
int a = 1 + 1

expression: term (('-' | '+') term)*
term: factor (('/' | '*') factor)*
factor: NUMBER | '(' expression ')'
eg.
5 - 3 * 2

======================

2 < 3 && ( 4 < 5 )

logic = [ and, or ]
compare = [ ==, !=, <=, >= ]

expression: base | BOOL (logic base | BOOL)?
base: INT compare INT | '(' expression ')'
